SELECT
  NON EMPTY {
    [Measures].[Total scores],
    [Measures].[contributors.COUNT],
    [Measures].[Average score per game]
  } ON COLUMNS,
  NON EMPTY Hierarchize(
    Union(
      [Games].[Games].[Team1Name].Members,
      Descendants(
        {
          [Games].[Games].[ALL].[AllMember].[Argentina]
        },
        [Games].[Games].[Team2Name]
      )
    )
  ) ON ROWS
  FROM [NanoPivotCube]
  CELL PROPERTIES VALUE, FORMATTED_VALUE, BACK_COLOR, FORE_COLOR, FONT_FLAGS
